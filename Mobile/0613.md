import 'package:flutter/material.dart';
import 'package:shared_preferences/shared_preferences.dart';
import 'dart:convert';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: '期末考試',
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  List<Map<String, dynamic>> items = [];

  @override
  void initState() {
    super.initState();
    _loadItems();
  }

  void _loadItems() async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    String? itemsString = prefs.getString('items');
    if (itemsString != null) {
      setState(() {
        items = List<Map<String, dynamic>>.from(json.decode(itemsString));
      });
    }
  }

  void _saveItems() async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    prefs.setString('items', json.encode(items));
  }

  void _addItem(Map<String, dynamic> newItem) {
    setState(() {
      items.add(newItem);
      _saveItems();
    });
  }

  void _editItem(int index, Map<String, dynamic> editedItem) {
    setState(() {
      items[index] = editedItem;
      _saveItems();
    });
  }

  void _deleteItem(int index) {
    setState(() {
      items.removeAt(index);
      _saveItems();
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('4110E113 陳威任'),
        actions: [
          TextButton(
            onPressed: () async {
              final newItem = await Navigator.push(
                context,
                MaterialPageRoute(
                  builder: (context) => EditItemPage(),
                ),
              );
              if (newItem != null) {
                _addItem(newItem);
              }
            },
            child: Text('新增', style: TextStyle(color: Colors.white)),
          ),
        ],
      ),
      body: ListView.builder(
        itemCount: items.length,
        itemBuilder: (context, index) {
          return ListTile(
            title: Text('${items[index]['name']} ${items[index]['category']} ${items[index]['price']}'),
            trailing: Row(
              mainAxisSize: MainAxisSize.min,
              children: [
                IconButton(
                  icon: Image.asset('assets/edit.png'),
                  onPressed: () async {
                    final editedItem = await Navigator.push(
                      context,
                      MaterialPageRoute(
                        builder: (context) => EditItemPage(item: items[index]),
                      ),
                    );
                    if (editedItem != null) {
                      _editItem(index, editedItem);
                    }
                  },
                ),
                IconButton(
                  icon: Image.asset('assets/delete.png'),
                  onPressed: () {
                    _deleteItem(index);
                  },
                ),
              ],
            ),
          );
        },
      ),
    );
  }
}

class EditItemPage extends StatefulWidget {
  final Map<String, dynamic>? item;

  EditItemPage({this.item});

  @override
  _EditItemPageState createState() => _EditItemPageState();
}

class _EditItemPageState extends State<EditItemPage> {
  final _formKey = GlobalKey<FormState>();
  final _nameController = TextEditingController();
  final _priceController = TextEditingController();
  String _category = '文具';

  @override
  void initState() {
    super.initState();
    if (widget.item != null) {
      _nameController.text = widget.item!['name'];
      _priceController.text = widget.item!['price'].toString();
      _category = widget.item!['category'];
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('4110E113 陳威任'),
      ),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Form(
          key: _formKey,
          child: Column(
            children: [
              TextFormField(
                controller: _nameController,
                decoration: InputDecoration(labelText: '品名'),
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return '請輸入品名';
                  }
                  return null;
                },
              ),
              DropdownButtonFormField<String>(
                value: _category,
                decoration: InputDecoration(labelText: '類別'),
                items: ['文具', '生鮮'].map((category) {
                  return DropdownMenuItem(
                    value: category,
                    child: Text(category),
                  );
                }).toList(),
                onChanged: (value) {
                  setState(() {
                    _category = value!;
                  });
                },
              ),
              TextFormField(
                controller: _priceController,
                decoration: InputDecoration(labelText: '價格'),
                keyboardType: TextInputType.number,
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return '請輸入價格';
                  }
                  return null;
                },
              ),
              SizedBox(height: 20),
              ElevatedButton(
                onPressed: () {
                  if (_formKey.currentState!.validate()) {
                    final newItem = {
                      'name': _nameController.text,
                      'category': _category,
                      'price': int.parse(_priceController.text),
                    };
                    Navigator.pop(context, newItem);
                  }
                },
                child: Text('儲存'),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
