```php
<!DOCTYPE HTML>
<HTML lang="en">
<head>
	<meta charset="UTF-8">
	<meta name="viewport" content="width=device-width, initial-scale=1.0">
	<title> CSV 文件上傳與顯示</title>
</head>
<body>
	<h1>上傳並顯示CSV文件內容</h1>
	<form action="" method="post" enctype="multipart/form-data">
		<label for="file">選擇CSV文件:</lable>
		<input type="file" id="file" name="file" accept=".csv" required><br><br>
		<button type="submit" name="submit">上傳並顯示</button>
	</form>
	
	<?php
	if ($_SERVER["REQUEST_METHOD"] == "POST" && isset($_FILES["file"]) && isset($_POST["submit"])){
		$file=$_FILES["file"];
		// 檢查文件是否上傳成功
		if ($file["error"]  != UPLOAD_ERR_OK){
			echo "<p>文件上傳失敗</p>";
			exit;
		}
		// 確保上傳的是CSV文件
		$fileType=pathinfo($file["name"], PATHINFO_EXTENSION);
		if ($fileType!="csv"){
		echo "<p>請上傳一個CSV文件</p>";
		exit;
		}
		// 讀取CSV內容
		$filePath=$file["tmp_name"];
		if (($handle=fopen($filePath,"r"))!==false){
			echo "<h2>文件內容</h2>";
			echo "<table border='1'>";
			// 讀取CSV標頭
			if (($header=fgetcsv($handle,1000,","))!==false){
				echo "<tr>";
				foreach($header as $col){
					echo"<th>" . htmlspecialchars($col)."</th>";
				}//foreach()
				echo "</tr>";
			}//if
			//讀取CSV每一行數據
			while (($data=fgetcsv($handle, 1000,",")) !==false){
				echo "<tr>";
				foreach($data as $col){
					echo"<td>" . htmlspecialchars($col)."</td>";
			}//foreach
			echo "</tr>";
		}//while()
		echo "</table>";
		fclose($handle);
	} else{
		echo"<p>讀取文件有誤</p>";
	}// if($handle)
	} else{
		echo "<p>等待輸入</p>";
	}//($SERVER)
	?>
</body>
</html>
```
